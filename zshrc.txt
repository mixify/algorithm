# for studying algorithm
algo () {
    echo "add, commit, and push to algorithm git"
    echo "also copy code to clip board"
    cat $1.cpp | xclip -sel clip
    git add ~/algorithm
    git commit -m "$1 algorithm solve comment : $2"
    git push origin master
}
